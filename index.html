Interpolation Graph


    var names = ['centripetal', 'chordal', 'uniform', 'disabled'];
    var groups = new vis.DataSet();
    groups.add({
        id: 0,
        content: names[0],
        options: {
            drawPoints: false,
            interpolation: {
                parametrization: 'centripetal'
            }
        }});

    groups.add({
        id: 1,
        content: names[1],
        options: {
            drawPoints: false,
            interpolation: {
                parametrization: 'chordal'
            }
        }});

    groups.add({
        id: 2,
        content: names[2],
        options: {
            drawPoints: false,
            interpolation: {
                parametrization: 'uniform'
            }
        }
    });

    groups.add({
        id: 3,
        content: names[3],
        options: {
            drawPoints: { style: 'circle' },
            interpolation: false
        }});

    var container3 = document.getElementById('visualization3');
    var dataset3 = new vis.DataSet();
    for (var i = 0; i < names.length; i++) {
        dataset3.add( [
            {x: '2014-06-12', y: 0 , group: i},
            {x: '2014-06-13', y: 40, group: i},
            {x: '2014-06-14', y: 10, group: i},
            {x: '2014-06-15', y: 15, group: i},

            {x: '2014-06-20', y: 10, group: i},
            {x: '2014-06-21', y: 20, group: i}
        ]);
    }

    var options3 = {
        drawPoints: false,
        dataAxis: {visible: false},
        legend: true,
        start: '2014-06-11',

    };
    var graph2d = new vis.Graph2d(container3, dataset3, groups, options3);
